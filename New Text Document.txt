  private List<List<string>> AddPreSufIndex(Excel.Worksheet ex_ws_layout_file, List<List<string>> NewDestListAllModel)
  {
      for (int o = 0; o < NewDestListAllModel.Count; o++)
      {
          int q = NewDestListAllModel[o].Count - 1;
          while (q > 0 && NewDestListAllModel[o][q] != ";")
          {
              if (q > 0)
              {
                  q--;
              }
              else
                  break;
          }
          if (q > 0)
          {
              string prefix = "";
              string suffix = "";
              int addAt = 1;
              List<string> ListSamePrefix = new List<string>();
              int p = 1;
              int lenghtDestListBefore = 1;
              if (q - lenghtDestListBefore > 0)
              {
                  prefix = NewDestListAllModel[o][p].Trim().Substring(0, 5);

                  while (q - 1 > 0)
                  {
                          if (prefix == NewDestListAllModel[o][p].Trim().Substring(0, 5))
                          {
                              if (NewDestListAllModel[o][p].Length == 11)
                              {
                                  suffix = NewDestListAllModel[o][p].Substring(8, 2); // Lấy hậu tố từ chuỗi 10 ký tự
                              }

                              ListSamePrefix.Add(NewDestListAllModel[o][p].Trim().Substring(5, 3));
                              NewDestListAllModel[o].RemoveAt(p);
                              q--;
                              continue;
                          }
                          else
                          {
                              ListSamePrefix.Sort();

                              for (int i = 0; i < ListSamePrefix.Count; i++)
                              {
                                  ListSamePrefix[i] = prefix + ListSamePrefix[i] + suffix; // Áp dụng hậu tố đã lấy
                              }

                              NewDestListAllModel[o].InsertRange(lenghtDestListBefore, ListSamePrefix);
                              lenghtDestListBefore = lenghtDestListBefore + ListSamePrefix.Count;
                              prefix = NewDestListAllModel[o][lenghtDestListBefore].Trim().Substring(0, 5);
                              p = p + lenghtDestListBefore - 1;
                              q = q + lenghtDestListBefore - 1;
                          }
                     
                          
                  }
              }
          }

      }

      return NewDestListAllModel;
  }
